CREATE TABLE Customers (
  Id INT PRIMARY KEY,
  FullName NVARCHAR (50) CHECK(LEN(FullName) <=50) NOT NULL,
  IdentityNumber CHAR (11) CHECK(LEN(FullName) =11) NOT NULL,
  PhoneNumber CHAR (9) CHECK(LEN(PhoneNumber) =11) NOT NULL,
  Email VARCHAR(50) CHECK (Email  LIKE '%@%')  NOT NULL,
  type CHAR (1) CHECK(type IN (0,1)) NOT NULL,
)
