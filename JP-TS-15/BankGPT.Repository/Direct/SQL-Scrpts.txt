CREATE TABLE Customers (
    id INT PRIMARY KEY,
    name VARCHAR(50) CHECK (LEN(name) <= 50) not null,
    identitynumber CHAR(11) CHECK (LEN(identitynumber) = 11) not null,
    phonenumber VARCHAR(15) CHECK (LEN(phonenumber) >= 10) not null,
    email VARCHAR(50) CHECK (email LIKE '%@%') not null,
    type INT CHECK (type IN (0, 1)) not null
);
